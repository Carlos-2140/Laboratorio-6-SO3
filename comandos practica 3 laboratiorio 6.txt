sudo dnf config-manager --set-enabled ol9_codeready_builder
sudo dnf install epel-release -y
sudo dnf update -y

sudo dnf install -y gcc gcc-c++ make cmake automake autoconf libtool git vim

sudo dnf install -y \
libpcap-devel \
pcre-devel \
libdnet-devel \
hwloc-devel \
openssl-devel \
zlib-devel \
luajit-devel \
pkgconf \
libmnl-devel \
libunwind-devel

sudo dnf install -y hyperscan hyperscan-devel xz-devel libuuid-devel

mkdir ~/snort_src
cd ~/snort_src

git clone https://github.com/snort3/libdaq.git
cd libdaq
./bootstrap
./configure
make
sudo make install
sudo ldconfig

cd ~/snort_src
git clone https://github.com/snort3/snort3.git
cd snort3

export PKG_CONFIG_PATH=/usr/local/lib/pkgconfig:$PKG_CONFIG_PATH
export PKG_CONFIG_PATH=/usr/local/lib64/pkgconfig:$PKG_CONFIG_PATH

./configure_cmake.sh --prefix=/usr/local/snort

cd build
make -j$(nproc)
sudo make install

/usr/local/snort/bin/snort -V

sudo mkdir -p /usr/local/snort/rules
sudo mkdir -p /var/log/snort

sudo nano /usr/local/snort/rules/local.rules

alert icmp any any -> $HOME_NET any (msg:"ICMP Ping Detectado"; sid:1000001; rev:1;)
alert tcp any any -> $HOME_NET 21 (msg:"Acceso detectado al puerto FTP 21"; sid:1000002; rev:1;)
alert tcp any any -> $HOME_NET 22 (msg:"Acceso detectado al puerto SSH 22"; sid:1000003; rev:1;)
alert tcp any any -> $HOME_NET 80 (msg:"Acceso detectado al puerto HTTP 80"; sid:1000004; rev:1;)

ip a

sudo /usr/local/snort/bin/snort -c /usr/local/snort/etc/snort/snort.lua \
-R /usr/local/snort/rules/local.rules \
-i eth0 -A alert_fast
